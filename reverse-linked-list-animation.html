let prev = null;
let current = head;
while (current !== null) {
    // 1. 保存下一个节点
    const next = current.next;
    
    // 2. 反转当前节点的指向
    current.next = prev;
    
    // 3. 移动指针
    prev = current;
    current = next;
}
return prev;
